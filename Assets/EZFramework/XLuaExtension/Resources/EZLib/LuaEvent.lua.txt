local moduleName = ...;
local M = { };
M.__index = M;
----- begin module -----
function M:New()
    local t = { };
    setmetatable(t, self);
    t.events = { };
    return t;
end

function M:AddListener(eventKey, eventFunc)
    self.events[eventKey] = eventFunc;
end

function M:RemoveListener(eventKey)
    self.events[eventKey] = nil;
end

function M:Clear()
	self.events = { };
end

function M:Call(...)
    local params = { ...};
    for key, func in pairs(self.events) do
        func(key, table.unpack(params));
    end
end
----- end -----
return M;
